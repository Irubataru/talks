\newcommand{\peak}[4]{%
\foreach \z in {0.01,0.02,...,1}{
  \pgfmathparse{sqrt(-ln(\z)/#4)}
  \let\r=\pgfmathresult
  \pgfmathparse{100*\z*#3*0.25}
  \let\col=\pgfmathresult
  \tdplotdrawarc[color=ICantExpress,line width=(1pt*#3*0.25),opacity=0.5]{(#1,#2,#3*\z)}{\r}{0}{360}{}{}
  \tdplotdrawarc[color=Marty!\col!Tropiteal, line width=0.8pt*#3*0.25,opacity=0.5]{(#1,#2,#3*\z)}{\r}{0}{360}{}{}
}}
